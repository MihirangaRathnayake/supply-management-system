import React, { useEffect, useState, useMemo } from 'react';
import axios from 'axios';
import { FontAwesomeIcon } from '@fortawesome/react-fontawesome';
import {
    faBoxes,
    faSync,
    faSearch,
    faWarehouse,
    faDollarSign,
    faCubes,
    faBox,
    faExclamationTriangle,
    faCheckCircle,
    faAdjust,
    faTimes,
    faChartBar,
    faLayerGroup,
    faBell
} from '@fortawesome/free-solid-svg-icons';
import { useToast } from '../components/ToastContainer';
import { useAuth } from '../context/AuthContext';

const Inventory = () => {
    const { showToast } = useToast();
    const { user } = useAuth();
    const [inventory, setInventory] = useState([]);
    const [analytics, setAnalytics] = useState({});
    const [lowStock, setLowStock] = useState([]);
    const [loading, setLoading] = useState(true);
    const [activeTab, setActiveTab] = useState('overview');
    const [search, setSearch] = useState('');
    const [filterWarehouse, setFilterWarehouse] = useState('ALL');
    const [adjustDrawer, setAdjustDrawer] = useState(false);
    const [selectedItem, setSelectedItem] = useState(null);
    const [adjustForm, setAdjustForm] = useState({ quantity: '', reason: '', type: 'ADJUSTMENT' });

    const canManage = ['ADMIN', 'MANAGER'].includes(user?.role);

    useEffect(() => {
        fetchData();
    }, []);

    const fetchData = async () => {
        setLoading(true);
        try {
            const [invRes, analyticsRes, lowStockRes] = await Promise.all([
                axios.get('/api/inventory'),
                axios.get('/api/inventory/analytics'),
                axios.get('/api/inventory/low-stock')
            ]);
            setInventory(invRes.data.data || []);
            setAnalytics(analyticsRes.data.data || {});
            setLowStock(lowStockRes.data.data || []);
        } catch (err) {
            console.error('Failed to load inventory', err);
            showToast('Unable to load inventory data', 'error', 4000);
        } finally {
            setLoading(false);
        }
    };

    const handleSeedData = async () => {
        if (!user || user.role !== 'ADMIN') {
            showToast('Only admin can seed inventory data', 'error', 4000);
            return;
        }
        try {
            const res = await axios.post('/api/inventory/seed');
            showToast(`Inventory seeded: ${res.data.data.created} records created`, 'success');
            await fetchData();
        } catch (err) {
            console.error('Failed to seed inventory', err);
            showToast(err.response?.data?.message || 'Failed to seed inventory', 'error', 4000);
        }
    };

    const normalizeInventory = (item) => {
        const qtyOnHand = Number(item.QUANTITY_ON_HAND || item.quantity_on_hand || 0);
        const qtyReserved = Number(item.QUANTITY_RESERVED || item.quantity_reserved || 0);
        const minStock = Number(item.MIN_STOCK_LEVEL || item.min_stock_level || 0);
        const unitPrice = Number(item.UNIT_PRICE || item.unit_price || 0);

        let status = 'HEALTHY';
        if (qtyOnHand === 0) status = 'OUT_OF_STOCK';
        else if (qtyOnHand <= minStock) status = 'CRITICAL';
        else if (qtyOnHand <= minStock * 1.5) status = 'LOW';

        return {
            id: item.INVENTORY_ID || item.inventory_id,
            productName: item.PRODUCT_NAME || item.product_name || 'Unknown',
            sku: item.SKU || item.sku || 'N/A',
            warehouseName: item.WAREHOUSE_NAME || item.warehouse_name || 'Unknown',
            city: item.CITY || item.city || '',
            qtyOnHand,
            qtyReserved,
            qtyAvailable: qtyOnHand - qtyReserved,
            minStock,
            unitPrice,
            totalValue: qtyOnHand * unitPrice,
            status
        };
    };

    const warehouses = useMemo(() => {
        const unique = [...new Set(inventory.map(i => i.WAREHOUSE_NAME || i.warehouse_name))];
        return unique.filter(Boolean);
    }, [inventory]);

    const filteredInventory = useMemo(() => {
        const term = search.trim().toLowerCase();
        let list = inventory.map(normalizeInventory);

        if (term) {
            list = list.filter(item =>
                [item.productName, item.sku, item.warehouseName].some(v => v.toLowerCase().includes(term))
            );
        }

        if (filterWarehouse !== 'ALL') {
            list = list.filter(item => item.warehouseName === filterWarehouse);
        }

        return list;
    }, [inventory, search, filterWarehouse]);

    const openAdjustDrawer = (item) => {
        setSelectedItem(item);
        setAdjustForm({ quantity: '', reason: '', type: 'ADJUSTMENT' });
        setAdjustDrawer(true);
    };

    const closeAdjustDrawer = () => {
        setAdjustDrawer(false);
        setSelectedItem(null);
    };

    const handleAdjustSubmit = async (e) => {
        e.preventDefault();
        if (!canManage) {
            showToast('Only admin or manager can adjust inventory', 'error', 4000);
            return;
        }
        try {
            await axios.post('/api/inventory/adjust', {
                inventoryId: selectedItem.id,
                adjustmentQty: Number(adjustForm.quantity),
                reason: adjustForm.reason,
                transactionType: adjustForm.type
            });
            showToast('Inventory adjusted successfully', 'success');
            closeAdjustDrawer();
            await fetchData();
        } catch (err) {
            console.error('Failed to adjust inventory', err);
            showToast(err.response?.data?.message || 'Failed to adjust inventory', 'error', 4000);
        }
    };

    const getStatusBadge = (status) => {
        const configs = {
            HEALTHY: { label: 'Healthy', color: 'bg-green-100 text-green-700', icon: faCheckCircle },
            LOW: { label: 'Low Stock', color: 'bg-amber-100 text-amber-700', icon: faExclamationTriangle },
            CRITICAL: { label: 'Critical', color: 'bg-red-100 text-red-700', icon: faExclamationTriangle },
            OUT_OF_STOCK: { label: 'Out of Stock', color: 'bg-slate-100 text-slate-700', icon: faTimes }
        };
        const config = configs[status] || configs.HEALTHY;
        return (
            <span className={`px-2.5 py-1 rounded-full text-xs font-semibold ${config.color} inline-flex items-center gap-1`}>
                <FontAwesomeIcon icon={config.icon} className="text-[10px]" />
                {config.label}
            </span>
        );
    };

    if (loading) {
        return (
            <div className="flex items-center justify-center min-h-screen">
                <div className="text-center">
                    <FontAwesomeIcon icon={faSync} className="text-5xl text-indigo-600 mb-4 animate-spin" />
                    <p className="text-lg font-semibold text-slate-700">Loading inventory...</p>
                </div>
            </div>
        );
    }

    return (
        <div className="space-y-6">
            {/* Header */}
            <div className="bg-gradient-to-r from-blue-600 via-indigo-600 to-purple-600 text-white rounded-2xl p-6 shadow-lg">
                <div className="flex items-center justify-between">
                    <div>
                        <h1 className="text-3xl font-bold mb-1">Inventory Management</h1>
                        <p className="text-sm text-white/80">Real-time stock monitoring and control</p>
                    </div>
                    <div className="flex gap-2">
                        {user?.role === 'ADMIN' && inventory.length === 0 && (
                            <button
                                onClick={handleSeedData}
                                className="px-4 py-2 bg-green-500 hover:bg-green-600 rounded-lg font-semibold transition-colors"
                            >
                                <FontAwesomeIcon icon={faBoxes} className="mr-2" />
                                Seed Data
                            </button>
                        )}
                        <button
                            onClick={fetchData}
                            className="px-4 py-2 bg-white/20 hover:bg-white/30 rounded-lg transition-colors"
                        >
                            <FontAwesomeIcon icon={faSync} className="mr-2" />
                            Refresh
                        </button>
                    </div>
                </div>
            </div>

            {/* Empty State */}
            {inventory.length === 0 ? (
                <div className="card text-center py-12">
                    <FontAwesomeIcon icon={faBoxes} className="text-6xl text-slate-300 mb-4" />
                    <h3 className="text-2xl font-bold text-slate-900 mb-2">No Inventory Data</h3>
                    <p className="text-slate-600 mb-6">
                        Create products and warehouses first, then seed inventory data.
                    </p>
                    {user?.role === 'ADMIN' && (
                        <button
                            onClick={handleSeedData}
                            className="px-6 py-3 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700 font-semibold"
                        >
                            <FontAwesomeIcon icon={faBoxes} className="mr-2" />
                            Seed Inventory Data
                        </button>
                    )}
                </div>
            ) : (
                <>
                    {/* Stats */}
                    <div className="grid grid-cols-1 md:grid-cols-4 gap-4">
                        <div className="card bg-gradient-to-br from-blue-500 to-blue-600 text-white">
                            <div className="flex items-center justify-between">
                                <div>
                                    <p className="text-xs uppercase font-semibold opacity-80">Total Items</p>
                                    <h3 className="text-3xl font-bold mt-1">{analytics.TOTAL_ITEMS || 0}</h3>
                                </div>
                                <FontAwesomeIcon icon={faCubes} className="text-4xl opacity-20" />
                            </div>
                        </div>
                        <div className="card bg-gradient-to-br from-indigo-500 to-indigo-600 text-white">
                            <div className="flex items-center justify-between">
                                <div>
                                    <p className="text-xs uppercase font-semibold opacity-80">Total Quantity</p>
                                    <h3 className="text-3xl font-bold mt-1">{(analytics.TOTAL_QUANTITY || 0).toLocaleString()}</h3>
                                </div>
                                <FontAwesomeIcon icon={faBox} className="text-4xl opacity-20" />
                            </div>
                        </div>
                        <div className="card bg-gradient-to-br from-purple-500 to-purple-600 text-white">
                            <div className="flex items-center justify-between">
                                <div>
                                    <p className="text-xs uppercase font-semibold opacity-80">Total Value</p>
                                    <h3 className="text-3xl font-bold mt-1">${((analytics.TOTAL_VALUE || 0) / 1000).toFixed(1)}K</h3>
                                </div>
                                <FontAwesomeIcon icon={faDollarSign} className="text-4xl opacity-20" />
                            </div>
                        </div>
                        <div className="card bg-gradient-to-br from-amber-500 to-amber-600 text-white">
                            <div className="flex items-center justify-between">
                                <div>
                                    <p className="text-xs uppercase font-semibold opacity-80">Low Stock</p>
                                    <h3 className="text-3xl font-bold mt-1">{analytics.LOW_STOCK_COUNT || 0}</h3>
                                </div>
                                <FontAwesomeIcon icon={faExclamationTriangle} className="text-4xl opacity-20" />
                            </div>
                        </div>
                    </div>

                    {/* Tabs */}
                    <div className="card">
                        <div className="flex border-b border-slate-200 mb-4">
                            {[
                                { id: 'overview', label: 'All Items', icon: faLayerGroup },
                                { id: 'alerts', label: 'Alerts', icon: faBell, badge: lowStock.length }
                            ].map(tab => (
                                <button
                                    key={tab.id}
                                    onClick={() => setActiveTab(tab.id)}
                                    className={`px-6 py-3 font-semibold transition-all flex items-center gap-2 ${
                                        activeTab === tab.id
                                            ? 'text-indigo-600 border-b-2 border-indigo-600'
                                            : 'text-slate-600 hover:text-slate-900'
                                    }`}
                                >
                                    <FontAwesomeIcon icon={tab.icon} />
                                    {tab.label}
                                    {tab.badge > 0 && (
                                        <span className="ml-1 px-2 py-0.5 bg-red-500 text-white text-xs rounded-full">
                                            {tab.badge}
                                        </span>
                                    )}
                                </button>
                            ))}
                        </div>

                        {/* Overview Tab */}
                        {activeTab === 'overview' && (
                            <div className="space-y-4">
                                {/* Filters */}
                                <div className="flex gap-3">
                                    <div className="relative flex-1">
                                        <FontAwesomeIcon icon={faSearch} className="absolute left-3 top-1/2 -translate-y-1/2 text-slate-400" />
                                        <input
                                            type="text"
                                            value={search}
                                            onChange={(e) => setSearch(e.target.value)}
                                            placeholder="Search products, SKU, warehouse..."
                                            className="w-full pl-10 pr-4 py-2 border border-slate-200 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent"
                                        />
                                    </div>
                                    <select
                                        value={filterWarehouse}
                                        onChange={(e) => setFilterWarehouse(e.target.value)}
                                        className="px-4 py-2 border border-slate-200 rounded-lg focus:ring-2 focus:ring-indigo-500"
                                    >
                                        <option value="ALL">All Warehouses</option>
                                        {warehouses.map(w => <option key={w} value={w}>{w}</option>)}
                                    </select>
                                </div>

                                {/* Table */}
                                <div className="overflow-x-auto">
                                    <table className="min-w-full">
                                        <thead className="bg-slate-50 border-b border-slate-200">
                                            <tr>
                                                <th className="px-6 py-3 text-left text-xs font-bold text-slate-700 uppercase">Product</th>
                                                <th className="px-6 py-3 text-left text-xs font-bold text-slate-700 uppercase">Warehouse</th>
                                                <th className="px-6 py-3 text-center text-xs font-bold text-slate-700 uppercase">On Hand</th>
                                                <th className="px-6 py-3 text-center text-xs font-bold text-slate-700 uppercase">Reserved</th>
                                                <th className="px-6 py-3 text-center text-xs font-bold text-slate-700 uppercase">Available</th>
                                                <th className="px-6 py-3 text-center text-xs font-bold text-slate-700 uppercase">Value</th>
                                                <th className="px-6 py-3 text-center text-xs font-bold text-slate-700 uppercase">Status</th>
                                                <th className="px-6 py-3 text-center text-xs font-bold text-slate-700 uppercase">Actions</th>
                                            </tr>
                                        </thead>
                                        <tbody className="divide-y divide-slate-100">
                                            {filteredInventory.length ? (
                                                filteredInventory.map((item) => (
                                                    <tr key={item.id} className="hover:bg-slate-50">
                                                        <td className="px-6 py-4">
                                                            <div className="font-semibold text-slate-900">{item.productName}</div>
                                                            <div className="text-xs text-slate-500">SKU: {item.sku}</div>
                                                        </td>
                                                        <td className="px-6 py-4">
                                                            <div className="flex items-center gap-2">
                                                                <FontAwesomeIcon icon={faWarehouse} className="text-slate-400" />
                                                                <div>
                                                                    <div className="text-sm font-medium">{item.warehouseName}</div>
                                                                    <div className="text-xs text-slate-500">{item.city}</div>
                                                                </div>
                                                            </div>
                                                        </td>
                                                        <td className="px-6 py-4 text-center">
                                                            <span className="text-lg font-bold">{item.qtyOnHand}</span>
                                                        </td>
                                                        <td className="px-6 py-4 text-center">
                                                            <span className="text-sm text-amber-600 font-semibold">{item.qtyReserved}</span>
                                                        </td>
                                                        <td className="px-6 py-4 text-center">
                                                            <span className="text-sm text-green-600 font-semibold">{item.qtyAvailable}</span>
                                                        </td>
                                                        <td className="px-6 py-4 text-center">
                                                            <div className="text-sm font-semibold">${item.totalValue.toFixed(2)}</div>
                                                        </td>
                                                        <td className="px-6 py-4 text-center">
                                                            {getStatusBadge(item.status)}
                                                        </td>
                                                        <td className="px-6 py-4 text-center">
                                                            <button
                                                                onClick={() => openAdjustDrawer(item)}
                                                                disabled={!canManage}
                                                                className="px-3 py-1.5 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700 disabled:opacity-50 text-sm"
                                                            >
                                                                <FontAwesomeIcon icon={faAdjust} className="mr-1" />
                                                                Adjust
                                                            </button>
                                                        </td>
                                                    </tr>
                                                ))
                                            ) : (
                                                <tr>
                                                    <td colSpan={8} className="px-6 py-12 text-center text-slate-500">
                                                        No inventory items found
                                                    </td>
                                                </tr>
                                            )}
                                        </tbody>
                                    </table>
                                </div>
                            </div>
                        )}

                        {/* Alerts Tab */}
                        {activeTab === 'alerts' && (
                            <div className="space-y-3">
                                {lowStock.length ? (
                                    lowStock.map((item) => {
                                        const normalized = normalizeInventory(item);
                                        return (
                                            <div
                                                key={item.INVENTORY_ID || item.inventory_id}
                                                className="p-4 rounded-lg border-2 border-red-200 bg-red-50 hover:bg-red-100"
                                            >
                                                <div className="flex items-center justify-between">
                                                    <div className="flex-1">
                                                        <div className="flex items-center gap-3 mb-2">
                                                            <FontAwesomeIcon icon={faExclamationTriangle} className="text-red-600" />
                                                            <div>
                                                                <h4 className="font-bold text-slate-900">{normalized.productName}</h4>
                                                                <p className="text-xs text-slate-600">SKU: {normalized.sku} â€¢ {normalized.warehouseName}</p>
                                                            </div>
                                                        </div>
                                                        <div className="flex items-center gap-4 text-sm">
                                                            <div>
                                                                <span className="text-slate-600">Current: </span>
                                                                <span className="font-bold text-red-600">{normalized.qtyOnHand}</span>
                                                            </div>
                                                            <div>
                                                                <span className="text-slate-600">Min Level: </span>
                                                                <span className="font-semibold">{normalized.minStock}</span>
                                                            </div>
                                                        </div>
                                                    </div>
                                                    <button
                                                        onClick={() => openAdjustDrawer(normalized)}
                                                        disabled={!canManage}
                                                        className="px-4 py-2 bg-red-600 text-white rounded-lg hover:bg-red-700 disabled:opacity-50 font-medium"
                                                    >
                                                        Restock
                                                    </button>
                                                </div>
                                            </div>
                                        );
                                    })
                                ) : (
                                    <div className="text-center py-12">
                                        <FontAwesomeIcon icon={faCheckCircle} className="text-5xl text-green-500 mb-3" />
                                        <p className="text-slate-600 font-medium">All inventory levels are healthy!</p>
                                    </div>
                                )}
                            </div>
                        )}
                    </div>
                </>
            )}

            {/* Adjust Drawer */}
            {adjustDrawer && selectedItem && (
                <div className="fixed inset-0 z-50 flex items-center justify-end bg-black/50 backdrop-blur-sm">
                    <div className="bg-white w-full max-w-lg h-full overflow-y-auto shadow-2xl">
                        <div className="sticky top-0 bg-gradient-to-r from-indigo-600 to-purple-600 text-white p-6 z-10">
                            <div className="flex items-start justify-between">
                                <div>
                                    <h2 className="text-2xl font-bold">Adjust Inventory</h2>
                                    <p className="text-sm text-white/80">Modify stock levels</p>
                                </div>
                                <button
                                    onClick={closeAdjustDrawer}
                                    className="w-8 h-8 rounded-lg bg-white/20 hover:bg-white/30 flex items-center justify-center"
                                >
                                    <FontAwesomeIcon icon={faTimes} />
                                </button>
                            </div>
                        </div>

                        <div className="p-6 space-y-6">
                            {/* Item Info */}
                            <div className="bg-slate-50 rounded-lg p-4 border border-slate-200">
                                <h3 className="font-bold text-slate-900 mb-1">{selectedItem.productName}</h3>
                                <p className="text-xs text-slate-600 mb-3">SKU: {selectedItem.sku}</p>
                                <div className="grid grid-cols-2 gap-3 text-sm">
                                    <div className="bg-white rounded-lg p-3 border">
                                        <div className="text-xs text-slate-600 mb-1">Current Stock</div>
                                        <div className="text-2xl font-bold">{selectedItem.qtyOnHand}</div>
                                    </div>
                                    <div className="bg-white rounded-lg p-3 border">
                                        <div className="text-xs text-slate-600 mb-1">Available</div>
                                        <div className="text-2xl font-bold text-green-600">{selectedItem.qtyAvailable}</div>
                                    </div>
                                </div>
                            </div>

                            {!canManage && (
                                <div className="rounded-lg border border-amber-200 bg-amber-50 text-amber-800 px-4 py-3 text-sm">
                                    <FontAwesomeIcon icon={faExclamationTriangle} className="mr-2" />
                                    Only admin or manager can adjust inventory.
                                </div>
                            )}

                            <form onSubmit={handleAdjustSubmit} className="space-y-5">
                                <div>
                                    <label className="block text-sm font-bold text-slate-700 mb-2">
                                        Transaction Type <span className="text-red-500">*</span>
                                    </label>
                                    <select
                                        value={adjustForm.type}
                                        onChange={(e) => setAdjustForm({ ...adjustForm, type: e.target.value })}
                                        required
                                        className="w-full px-4 py-3 border border-slate-200 rounded-lg focus:ring-2 focus:ring-indigo-500"
                                    >
                                        <option value="ADJUSTMENT">Manual Adjustment</option>
                                        <option value="RECEIPT">Receipt</option>
                                        <option value="ISSUE">Issue</option>
                                        <option value="RETURN">Return</option>
                                        <option value="WRITE_OFF">Write Off</option>
                                    </select>
                                </div>

                                <div>
                                    <label className="block text-sm font-bold text-slate-700 mb-2">
                                        Adjustment Quantity <span className="text-red-500">*</span>
                                    </label>
                                    <input
                                        type="number"
                                        value={adjustForm.quantity}
                                        onChange={(e) => setAdjustForm({ ...adjustForm, quantity: e.target.value })}
                                        required
                                        className="w-full px-4 py-3 border border-slate-200 rounded-lg focus:ring-2 focus:ring-indigo-500"
                                        placeholder="Enter quantity (use negative for decrease)"
                                    />
                                    <p className="mt-2 text-xs text-slate-600">
                                        Use positive numbers to increase stock, negative to decrease
                                    </p>
                                    {adjustForm.quantity && (
                                        <div className="mt-3 p-3 bg-indigo-50 rounded-lg border border-indigo-200">
                                            <div className="text-sm font-semibold text-indigo-900">
                                                New Stock Level: {selectedItem.qtyOnHand + Number(adjustForm.quantity)}
                                            </div>
                                        </div>
                                    )}
                                </div>

                                <div>
                                    <label className="block text-sm font-bold text-slate-700 mb-2">
                                        Reason <span className="text-red-500">*</span>
                                    </label>
                                    <textarea
                                        value={adjustForm.reason}
                                        onChange={(e) => setAdjustForm({ ...adjustForm, reason: e.target.value })}
                                        required
                                        rows={3}
                                        className="w-full px-4 py-3 border border-slate-200 rounded-lg focus:ring-2 focus:ring-indigo-500"
                                        placeholder="Explain the reason for this adjustment..."
                                    />
                                </div>

                                <div className="flex gap-3 pt-4 border-t">
                                    <button
                                        type="button"
                                        onClick={closeAdjustDrawer}
                                        className="flex-1 px-4 py-3 border border-slate-200 rounded-lg text-slate-700 hover:bg-slate-50 font-semibold"
                                    >
                                        Cancel
                                    </button>
                                    <button
                                        type="submit"
                                        disabled={!canManage}
                                        className="flex-1 px-4 py-3 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700 disabled:opacity-50 font-semibold"
                                    >
                                        <FontAwesomeIcon icon={faCheckCircle} className="mr-2" />
                                        Apply
                                    </button>
                                </div>
                            </form>
                        </div>
                    </div>
                </div>
            )}
        </div>
    );
};

export default Inventory;
